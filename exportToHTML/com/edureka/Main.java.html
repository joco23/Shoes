class Shoe {
    String type;
    String color;
    String brand;
    int number;
    int price;

    public Shoe() {

    }

        this.type=type;
        this.color=color;

        this.price=price;
    }
    void showShoeDetails(){

        System.out.println("--Shoe brand : "+brand);
        System.out.println("--Shoe number : "+number);
        System.out.println("--Shoe price : "+price);
        System.out.println("-------------------");
    }
